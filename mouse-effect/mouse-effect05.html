<script>
const cursor = $(".cursor");
    const follwer = $(".cursor-follower");
    let x = 0;
    let y = 0;
    let newX = 0;
    let newY = 0;
    let newX2 = 0;
    let newY2 = 0;
    let fmouseX = 0;
    let fmouseY = 0;

    $(".move-image").mousemove(function(e){
        //원형 마우스 움직이기
        gsap.to(cursor,0.4,{left :e.pageX -(cursor.width()/2), top:e.pageY - (cursor.height()/2)})    
        gsap.to(follwer,0.8,{left :e.pageX-(follwer.width()/2), top:e.pageY - (follwer.height()/2)})

        //좌표값 중앙을 0으로 초기화
        x = e.pageX;
        y = e.pageY;
        newX = x - $(window).width()/2;
        newY = y - $(window).height()/2;
        $(".mouseX").text(newX);
        $(".mouseY").text(newX);

        // let xX = e.clientX -($(".move-image").width()/2 + $(".move-image").position().left);
        // let yY = e.clientY -($(".move-image").height()/2 + $(".move-image").position().top);

        const moveImg = document.querySelector(".move-image");
    
        //move-image의 중앙 
        moveImg.addEventListener("mousemove",function(e){
        const rect = this.getBoundingClientRect();
        console.log(rect);
        this.style.setProperty("--x",e.clientX - ((rect.width/2)+rect.left));
        this.style.setProperty("--y",e.clientY - ((rect.height/2)+rect.top));
        console.log("e.clientX"+e.clientX);
        console.log("e.clientY"+e.clientY);
        console.log("(rect.width/2)+rect.left)"+rect.width/2+rect.left);
        console.log("(rect.height/2)+rect.top)"+rect.height/2+rect.top);
        console.log("결과x:"+ e.clientX - ((rect.width/2)+rect.left));
        console.log("결과y:"+ e.clientY - ((rect.height/2)+rect.top));

        gsap.to(".cursor", {duration: .2, left: e.pageX - 10, top: e.pageY - 10});

            })
            moveImg.addEventListener("mouseleave", function(){
            this.style.setProperty("--x", 0)
            this.style.setProperty("--y", 0)
            });
    })

    $(".btn2").click(function(){
        let btn2Display = $(".cursor .mouse").css("display");
        if(btn2Display == "none"){
            $(".cursor .mouse").css({"display":"block"});
            $(this).text("좌표 없애기");
        }else{
            $(".cursor .mouse").css({"display":"none"});
            $(this).text("좌표 보기");
        }
    })

    let body = document.querySelector("body, html");
    body.addEventListener("mousemove", function (event) {
        document.querySelector(".clientX").textContent = event.clientX;
        document.querySelector(".clientY").textContent = event.clientY;
        document.querySelector(".offsetX").textContent = event.offsetX;
        document.querySelector(".offsetY").textContent = event.offsetY;
        document.querySelector(".pageX").textContent = event.pageX;
        document.querySelector(".pageY").textContent = event.pageY;
        document.querySelector(".screenX").textContent = event.screenX;
        document.querySelector(".screenY").textContent = event.screenY;
    })

//modal 버튼, close버튼 클릭
$(".btn").click(function () {
    $("#modal").slideDown();
})
$(".close").click(function () {
    $("#modal").slideUp();
})
</script>
